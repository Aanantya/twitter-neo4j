Top Mentions Most Tagged 
MATCH
  (u:User {screen_name:'{{ user }}'})-[:POSTS]->(t:Tweet)-[:MENTIONS]->(m:User)
RETURN
  m.screen_name AS screen_name, COUNT(m.screen_name) AS count 
ORDER BY 
  count 
DESC LIMIT 10
 
MATCH
  (h:Hashtag)-[:TAGS]-gt;(t:Tweet)
WITH 
  h, COUNT(h) AS Hashtags
ORDER BY 
  Hashtags DESC
LIMIT 10
RETURN 
  h.name, Hashtags
